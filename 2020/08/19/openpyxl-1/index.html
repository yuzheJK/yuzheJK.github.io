openpyxl(1)

openpyxl简单使用

openpyxl

>>> from openpyxl import Workbook
>>> wb = Workbook()
>>> ws = wb.active
>>> type(wb)
<class 'openpyxl.workbook.workbook.Workbook'>
>>> type(ws)
<class 'openpyxl.worksheet.worksheet.Worksheet'>

查看已有的工作表

>>> from openpyxl import load_workbook
>>> wb = load_workbook(r'C:\Users\Mto\Desktop\test.xlsx')
>>> print(wb.sheetnames)
['Sheet', 'Mysheet']
>>> ws = wb.active
>>> ws.title
'Sheet'
>>> A1 = ws['A1']
>>> A1
<Cell 'Sheet'.A1>
>>> A1.value
'A1'
>>> cell_range = ws['A1':'C2']
>>> cell_range
((<Cell 'Sheet'.A1>, <Cell 'Sheet'.B1>, <Cell 'Sheet'.C1>), (<Cell 'Sheet'.A2>, <Cell 'Sheet'.B2>, <Cell 'Sheet'.C2>))
>>> type(A1)
<class 'openpyxl.cell.cell.Cell'>
>>> type(cell_range)
<class 'tuple'>
>>> for row in ws.values:
	for value in row:
		print(value)

Worksheet.iter_rows()函数

>>> for row in ws.iter_rows(min_row=1, max_col=3, max_row=2):
	for cell in row:
		print(cell)

<Cell 'Sheet'.A1>
<Cell 'Sheet'.B1>
<Cell 'Sheet'.C1>
<Cell 'Sheet'.A2>
<Cell 'Sheet'.B2>
<Cell 'Sheet'.C2>

Worksheet.iter_cols()函数

>>> for col in ws.iter_cols(min_row=1,max_col=3,max_row=2):
	for cell in col:
		print(cell)

<Cell 'Sheet'.A1>
<Cell 'Sheet'.A2>
<Cell 'Sheet'.B1>
<Cell 'Sheet'.B2>
<Cell 'Sheet'.C1>
<Cell 'Sheet'.C2>

两个 Worksheet.iter_rows() 和 Worksheet.iter_cols() 可以采取 values_only 仅返回单元格值的参数，出于性能原因，Worksheet.iter_rows() Worksheet.iter_cols() 方法在只读模式下不可用。

>>> for row in ws.iter_rows(min_row=1,max_col=3,max_row=2,values_only=True):
	print(row)
	
('A1', 'B1', 'C1')
('A2', 'B2', 'C2')

如果需要遍历文件的所有行或列，则可以使用 Worksheet.rows 属性或 Worksheet.columns 属性

>>> tuple(ws.rows)
>>> tuple(ws.columns)

出于性能原因，Worksheet.rows Worksheet.columns 属性在只读模式下不可用

创建工作表

from openpyxl import Workbook
wb=Workbook()#创建一个工作簿
ws=wb.active#获取工作的激活工作表
ws['A1']=42#在A1中插入内容
ws.append([1,2,3])#这里其实是在第二行插入了3个数字，占用了三个单元格
wb.save('sample.xlsx')#保存到当前目录
