mysqlclient简单使用

mysqlclient的简单使用

创建数据库表

import MySQLdb
conn = MySQLdb.connect(host = 'localhost',user='root',passwd='root',db = 'test')
cursor = conn.cursor()#获取游标
#如果已经存在该表则使用execute()方法删除
cursor.execute("drop table if exists conda")
#创建数据表SQL语句
sql = """create table conda(
        first_name char(10) not null,
        last_name char(10),
        age int,
        sex char(1))"""

cursor.execute(sql)
conn.close()

在进行数据库连接时将编码设置为中文

conn = MySQLdb.connect(host = 'localhost',user='root',passwd='root',db = 'test'，charset="utf-8")

项目中插入数据

cur.execute("insert into users (id,username,password) values (%s,%s,%s),(id, user, passwd)")

插入数据

import MySQLdb

conn = MySQLdb.connect(host = 'localhost',user='root',passwd='root',db = 'test')
cur = conn.cursor()#通过数据库连接conn下的cursor()方法创建游标
try:
    cur.execute("insert into users (id,username,password) values ('5','conda1','conda321')")
    cur.close() #关闭游标
    conn.commit()# 提交到数据库执行
except:
    conn.rollback()

conn.close()

rollback 回滚的意思。 就是数据库里做修改后 （ update ,insert , delete）未commit 之前 使用rollback 可以恢复数据到修改之前。

读取数据

import MySQLdb

conn = MySQLdb.connect(host = 'localhost' ,user = 'root' ,passwd = 'root' ,db = 'test')

cur = conn.cursor()

sql = "SELECT * FROM users WHERE id > 1"
try:
    cur.execute(sql)            # 执行sql语句
    results = cur.fetchall()    # 获取所有记录列表
    for row in results:
        id = row[0]
        username = row[1]
        password = row[2]
        print("id=%d, username=%s, password=%s" % (id, username, password))
except:
        print("Error: unable to fecth data")
        
conn.close() #关闭数据库
